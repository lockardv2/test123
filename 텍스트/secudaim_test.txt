#include <stdio.h>

int main() {
    double num1, num2;
    char op;

    printf("두 숫자를 입력하세요: ");
    scanf("%lf %lf", &num1, &num2);

    printf("연산자를 입력하세요 (+, -, *, /): ");
    scanf(" %c", &op);

    switch(op) {
        case '+':
            printf("%.2lf + %.2lf = %.2lf\n", num1, num2, num1 + num2);
            break;
        case '-':
            printf("%.2lf - %.2lf = %.2lf\n", num1, num2, num1 - num2);
            break;
        case '*':
            printf("%.2lf * %.2lf = %.2lf\n", num1, num2, num1 * num2);
            break;
        case '/':
            if (num2 != 0)
                printf("%.2lf / %.2lf = %.2lf\n", num1, num2, num1 / num2);
            else
                printf("0으로 나눌 수 없습니다.\n");
            break;
        default:
            printf("올바른 연산자가 아닙니다.\n");
    }

    return 0;
}
